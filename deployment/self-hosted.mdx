## Kubernetes Deployment

```bash
# Add Helm repository
helm repo add datarm https://charts.datarm.com
helm repo update

# Install AVA
helm install ava datarm/ava \
  --namespace ava \
  --create-namespace \
  --set domain=ava.your-domain.com \
  --set postgresql.auth.password=YOUR_PASSWORD
```

## Docker Compose

```yaml
version: '3.8'
services:
  ava:
    image: datarm/ava:latest
    ports:
      - "8080:8080"
    environment:
      - DATABASE_URL=postgresql://postgres:password@postgres:5432/ava
      - REDIS_URL=redis://redis:6379
    depends_on:
      - postgres
      - redis

  postgres:
    image: postgres:15
    environment:
      - POSTGRES_PASSWORD=password
      - POSTGRES_DB=ava
    volumes:
      - postgres_data:/var/lib/postgresql/data

  redis:
    image: redis:7-alpine
```
