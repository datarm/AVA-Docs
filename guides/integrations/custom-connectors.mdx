## Overview

Create custom connectors to integrate AVA with proprietary or unsupported data sources.

## Connector SDK

Install the AVA Connector SDK:

```bash
pip install ava-connector-sdk
```

## Building a Custom Connector

<Tabs>
  <Tab title="Python">
    ```python
    from ava.connectors import BaseConnector, ConnectionConfig

    class MyCustomConnector(BaseConnector):
        def __init__(self, config: ConnectionConfig):
            super().__init__(config)
            self.api_url = config.api_url
            self.api_key = config.api_key

        def test_connection(self) -> bool:
            """Test connection to data source"""
            try:
                response = requests.get(
                    f"{self.api_url}/health",
                    headers={"Authorization": f"Bearer {self.api_key}"}
                )
                return response.status_code == 200
            except Exception as e:
                self.log_error(f"Connection test failed: {e}")
                return False

        def discover_schema(self):
            """Discover available tables/collections"""
            response = requests.get(
                f"{self.api_url}/tables",
                headers={"Authorization": f"Bearer {self.api_key}"}
            )
            return self.parse_schema(response.json())

        def fetch_data(self, table_name: str, limit: int = 1000):
            """Fetch sample data from table"""
            response = requests.get(
                f"{self.api_url}/tables/{table_name}/data",
                params={"limit": limit},
                headers={"Authorization": f"Bearer {self.api_key}"}
            )
            return response.json()

        def get_metadata(self, table_name: str):
            """Get table metadata"""
            response = requests.get(
                f"{self.api_url}/tables/{table_name}/metadata",
                headers={"Authorization": f"Bearer {self.api_key}"}
            )
            return self.parse_metadata(response.json())
    ```

  <Tab title="JavaScript">
    ```javascript
    import { BaseConnector } from '@datarm/ava-connector-sdk';

    class MyCustomConnector extends BaseConnector {
      constructor(config) {
        super(config);
        this.apiUrl = config.apiUrl;
        this.apiKey = config.apiKey;
      }

      async testConnection() {
        try {
          const response = await fetch(`${this.apiUrl}/health`, {
            headers: { 'Authorization': `Bearer ${this.apiKey}` }
          });
          return response.ok;
        } catch (error) {
          this.logError(`Connection test failed: ${error}`);
          return false;
        }
      }

      async discoverSchema() {
        const response = await fetch(`${this.apiUrl}/tables`, {
          headers: { 'Authorization': `Bearer ${this.apiKey}` }
        });
        const data = await response.json();
        return this.parseSchema(data);
      }

      async fetchData(tableName, limit = 1000) {
        const response = await fetch(
          `${this.apiUrl}/tables/${tableName}/data?limit=${limit}`,
          { headers: { 'Authorization': `Bearer ${this.apiKey}` } }
        );
        return response.json();
      }
    }

    export default MyCustomConnector;
    ```

## Register Custom Connector

```bash
# Package connector
ava connectors package ./my-connector

# Register with AVA
ava connectors register \
  --name "My Custom Source" \
  --package my-connector-1.0.0.tar.gz \
  --icon custom-icon.png

# Use connector
ava connect my-custom-source \
  --config custom-config.yaml
```

## Connector Configuration

```yaml
connector:
  name: My Custom Connector
  version: 1.0.0
  type: custom

  config_schema:
    - name: api_url
      type: string
      required: true
      description: API endpoint URL

    - name: api_key
      type: string
      required: true
      secret: true
      description: API authentication key

  capabilities:
    - schema_discovery
    - data_sampling
    - metadata_extraction
    - incremental_sync

  scheduling:
    supported: true
    min_interval: 1h
    default_interval: 24h
```

## Testing Your Connector

```python
# Test connector locally
from ava.connectors.testing import ConnectorTester

tester = ConnectorTester(MyCustomConnector)

# Run test suite
results = tester.run_tests(
    config={
        "api_url": "https://api.example.com",
        "api_key": "test_key"
    }
)

print(f"Tests passed: {results.passed}/{results.total}")
```
