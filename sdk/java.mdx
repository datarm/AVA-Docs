## Installation

### Maven

```xml
<dependency>
  <groupId>com.datarm</groupId>
  <artifactId>ava-sdk</artifactId>
  <version>1.0.0</version>
</dependency>
```

### Gradle

```gradle
implementation 'com.datarm:ava-sdk:1.0.0'
```

## Quick Start

```java
import com.datarm.ava.Client;
import com.datarm.ava.model.DataSource;

// Initialize client
Client client = new Client("YOUR_API_KEY");

// List data sources
List<DataSource> sources = client.getDataSources().list();
for (DataSource source : sources) {
    System.out.println(source.getName() + ": " + source.getStatus());
}

// Query data catalog
QueryResults results = client.query("Find all PII in Salesforce");
for (DataAsset asset : results.getAssets()) {
    System.out.println(asset.getName() + " - Risk Score: " + asset.getRiskScore());
}
```

## Data Sources

```java
import com.datarm.ava.model.*;

// Create data source
DataSourceConfig config = DataSourceConfig.builder()
    .name("Production Snowflake")
    .type(DataSourceType.SNOWFLAKE)
    .config(Map.of(
        "account", "your-account.snowflakecomputing.com",
        "warehouse", "COMPUTE_WH",
        "database", "PROD_DB"
    ))
    .credentials(Map.of(
        "username", "ava_service",
        "password", "encrypted_password"
    ))
    .build();

DataSource source = client.getDataSources().create(config);

// Trigger scan
Scan scan = client.getDataSources().scan(source.getId(), ScanProfile.STANDARD);

// Get scan status
ScanStatus status = client.getDataSources().getScanStatus(scan.getId());
System.out.println("Status: " + status.getStatus() + ", Progress: " + status.getProgress() + "%");
```

## Policies

```java
// Create policy
PolicyConfig policyConfig = PolicyConfig.builder()
    .name("PII Protection Policy")
    .type(PolicyType.ACCESS_CONTROL)
    .scope(PolicyScope.builder()
        .classifications(List.of("PII", "PII_EMAIL"))
        .build())
    .rules(List.of(
        Rule.builder().type("require_encryption").enabled(true).build(),
        Rule.builder().type("require_mfa").enabled(true).build()
    ))
    .build();

Policy policy = client.getPolicies().create(policyConfig);

// Apply policy
client.getPolicies().apply(policy.getId());
```

## Error Handling

```java
import com.datarm.ava.exception.*;

try {
    DataSource source = client.getDataSources().get("ds_123");
} catch (NotFoundException e) {
    System.err.println("Data source not found");
} catch (RateLimitException e) {
    System.err.println("Rate limit exceeded. Retry after " + e.getRetryAfter() + " seconds");
} catch (AVAException e) {
    System.err.println("API error: " + e.getMessage());
}
```
