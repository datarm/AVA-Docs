## Quick Start

```python
from ava import Client

# Initialize client
client = Client(api_key="YOUR_API_KEY")

# List data sources
sources = client.data_sources.list()
for source in sources:
    print(f"{source.name}: {source.status}")

# Query data catalog
results = client.query("Find all PII in Salesforce")
for asset in results:
    print(f"{asset.name} - Risk Score: {asset.risk_score}")
```

## Authentication

```python
# API key authentication
client = Client(api_key="ava_sk_live_...")

# Environment variable
import os
client = Client(api_key=os.getenv("AVA_API_KEY"))

# Custom domain (your Azure tenant deployment)
client = Client(
    api_key="YOUR_API_KEY",
    base_url="https://ava.your-domain.com"
)
```

## Data Sources

```python
# Create data source
source = client.data_sources.create(
    name="Production Snowflake",
    type="snowflake",
    config={
        "account": "your-account.snowflakecomputing.com",
        "warehouse": "COMPUTE_WH",
        "database": "PROD_DB"
    },
    credentials={
        "username": "ava_service",
        "private_key_path": "/path/to/key.pem"
    }
)

# Trigger scan
scan = client.data_sources.scan(source.id, profile="standard")
print(f"Scan initiated: {scan.id}")

# Get scan status
status = client.data_sources.get_scan_status(scan.id)
print(f"Status: {status.status}, Progress: {status.progress}%")
```

## Policies

```python
# Create policy
policy = client.policies.create(
    name="PII Protection Policy",
    type="access_control",
    scope={
        "classifications": ["PII", "PII_EMAIL"]
    },
    rules=[
        {
            "type": "require_encryption",
            "enabled": True
        },
        {
            "type": "require_mfa",
            "enabled": True
        }
    ]
)

# Apply policy
client.policies.apply(policy.id)

# List violations
violations = client.policies.violations(
    policy_id=policy.id,
    severity=["high", "critical"]
)
```

## Risk Management

```python
# Get high-risk assets
high_risk = client.risk.list(
    level=["high", "critical"],
    limit=50
)

for asset in high_risk:
    print(f"{asset.name}")
    print(f"  Risk Score: {asset.risk_score}")
    print(f"  Factors: {', '.join([f.description for f in asset.risk_factors])}")

# Configure risk alerts
alert = client.risk.create_alert(
    name="Critical Risk Alert",
    condition="risk_score > 90",
    channels=["email", "slack"],
    recipients=["security-team@company.com"]
)
```

## Async Support

```python
import asyncio
from ava import AsyncClient

async def main():
    async with AsyncClient(api_key="YOUR_API_KEY") as client:
        # Concurrent requests
        sources, policies, users = await asyncio.gather(
            client.data_sources.list(),
            client.policies.list(),
            client.users.list()
        )

        print(f"Sources: {len(sources)}")
        print(f"Policies: {len(policies)}")
        print(f"Users: {len(users)}")

asyncio.run(main())
```

## Error Handling

```python
from ava import Client, AVAError, RateLimitError, NotFoundError

client = Client(api_key="YOUR_API_KEY")

try:
    source = client.data_sources.get("ds_123")
except NotFoundError:
    print("Data source not found")
except RateLimitError as e:
    print(f"Rate limit exceeded. Retry after {e.retry_after} seconds")
except AVAError as e:
    print(f"API error: {e.message}")
```
